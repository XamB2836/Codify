# Exemple en Python avec annotations de type
from typing import List

def fibonacci(n: int) -> List[int]:
    """Génère la séquence de Fibonacci"""
    sequence = [0, 1]
    while len(sequence) < n:
        sequence.append(sum(sequence[-2:]))
    return sequence[:n]

print(fibonacci(10))
